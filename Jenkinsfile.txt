pipeline{
    agent any
    stages{
        stage('ShellScript_KillProcess')
            {
            steps
                {
                sh '''
                #!/bin/bash
                #this script is to automate kill process
                arr=$(pgrep rasa)
                echo "$arr"
                IFS=$'\n'
                for i in $arr
                do
                    kill $i
                done
                conda init bash
                chmod a+rwx models
                rasa train
                nohup rasa run --enable-api --cors "*" --debug &
                nohup rasa run actions --debug &

                '''
                }
            }
      

        }
    }
