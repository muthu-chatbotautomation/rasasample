pipeline{
    agent any
    stages
        {
        
           stage('ShellScript_KillExistingProcess')
           {
                steps
                    {
                        sh '''
                        #!/bin/bash
                        #this script is to automate kill process
                        arr=$(pgrep rasa)
                        echo "$arr"
                        IFS=$'\n'
                        for i in $arr
                        do
                            kill $i
                        done
                        '''
                    }
             }


           stage('Rasa_Train')
           {
                steps
                    {
                        sh '''
                        #!/bin/bash
                        #this script is to automate kill process
                        conda init bash
                        chmod a+rwx models
                        rasa train
                        '''
                    }
             }
             
           stage('Rasa_Test')
           {
                steps
                    {
                        sh '''

                        #this script is to automate kill process

                        '''
                    }
             }

                    '''
                }
            }
